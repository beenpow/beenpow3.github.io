ll N;

    for(ll i = 2; i * i <= x ; i++){
        if(x % i == 0) return false;
    }
    return true;
}
void process(){
    vec.clear();
    cin >> N;
    if(N%2 == 0){
        cout << N/2 <<" " << N/2 << endl;
    }else{
        if(isPrime(N)){
            cout << "1" << " " << N-1 << endl;
        }else{
            for(ll i=1; i*i <= N; i++){
                if(N % i == 0){
                    vec.push_back(i);
                    if(i != N/i) vec.push_back(N/i);
                }
            }
            sort(all(vec));
            vec.pop_back();
            cout << vec.back() << " " << N - vec.back() << endl;
        }
    }
}
