# BOJ::14502 연구소
[BOJ] : <https://www.acmicpc.net/problem/14502>
- Level : Gold 5

# 시사점

- 시뮬레이션 + 백트랙킹 기본 문제입니다.

# 이해(6)

# 설계(4)

- bactrack을 통해 빈 공간 3 곳을 벽으로 변경합니다.
  - 변경된 벽을 기본으로 bfs 탐색합니다.

## 시간 복잡도

- 3개 고르는 백트래킹 (N^2) C 3 
- 3개를 고른 후, BFS : O(4 * N^2)

## 공간 복잡도

# 구현(20)

```cpp
#include<bits/stdc++.h>
using namespace std;
const int MAX_N = 8;
const int dx[]={-1, 0, 1, 0}, dy[]={0, 1, 0, -1};

vector<pair<int, int> > ovirus;
int n, m;
int mx0, ans;

bool over(int x, int y){return (x<0 || y<0 || x>=n || y>=m);}
int bfs(int oa[8][8]){
    int a[8][8]={0};
    for(int i = 0; i < MAX_N; i++)for(int j = 0; j < MAX_N; j++)a[i][j] = oa[i][j];

    int ret = mx0;
    queue<pair<int, int> > q;
    for(int i = 0; i < ovirus.size(); i++)
        q.push({ovirus[i].first, ovirus[i].second});
    while(!q.empty()){
        int x = q.front().first, y = q.front().second; q.pop();
        for(int i = 0; i < 4; i++){
            int nx = x+dx[i], ny = y+dy[i];
            if(over(nx, ny) || a[nx][ny] != 0) continue;
            ret -= 1;
            q.push({nx, ny});
            a[nx][ny] = 2;
        }
    }

    return ret;
}

void solve(int x, int y, int used, int a[8][8]){
    if(used == 3){
        int ret = bfs(a);
        ans = max(ans, ret);
        return;
    }
    if(y == m) x+=1, y = 0;
    if(x == n) return;
    for(int i = x; i < n; i++, y = 0)
        for(int j = y; j < m; j++){
            if(a[i][j] == 0){
                a[i][j] = 1;
                solve(i, j+1, used+1, a);
                a[i][j] = 0;
            }
        }
}
int main(){
    freopen("input.txt", "r", stdin);
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int oa[MAX_N][MAX_N]={0};
    cin >> n >> m;
    for(int i = 0; i < n ; i++)
        for(int j = 0; j < m; j++){
            cin >> oa[i][j];
            if(oa[i][j] == 0) mx0++;
            else if(oa[i][j] == 2) ovirus.push_back({i, j});
        }

    mx0 -= 3;
    solve(0, 0, 0, oa);
    cout << ans << endl;
    return 0;
}
```

```cpp
// O(N^2 * N^2 * N^2) * O(N^2)
#include<bits/stdc++.h>
#define endl '\n'
#define rep(i,a,b) for(int i=a;i<b;i++)
#define r_rep(i,a,b) for(int i=a;i>b;i--)
const int MAXNM = 8;
enum element {EMPTY=0, WALL, VIR};
const int dx[]={-1, 0, 1, 0}, dy[]={0, 1, 0, -1};
using namespace std;

int n, m, all0, ans;
int oa[MAXNM][MAXNM];
bool seen[MAXNM][MAXNM];
vector<pair<int,int> > virus;
void input(){
    cin >> n >> m;
    rep(i, 0, n) rep(j, 0, m){
        cin >> oa[i][j];
        if(oa[i][j] == 0) all0++;
        if(oa[i][j] == 2) virus.push_back({i, j});
    }
}
bool over(int x, int y){return (x<0 || y<0 || x>=n || y>=m);}
int bfs(const int (&a)[MAXNM][MAXNM]){
    int ret = all0;
    queue<pair<int,int> >q;
    rep(i, 0, virus.size()){
        q.push({virus[i].first, virus[i].second});
        seen[virus[i].first][virus[i].second] = true;
    }
    while(!q.empty()){
        int x = q.front().first, y = q.front().second; q.pop();
        rep(i, 0, 4){
            int nx = x+dx[i], ny = y+dy[i];
            if(over(nx, ny) || seen[nx][ny] || a[nx][ny] == WALL || a[nx][ny] == VIR) continue;
            seen[nx][ny] = true;
            q.push({nx, ny});
            ret--;
        }
    }
    return ret;
}
void backtrack(int x, int y, int cnt, int(&a)[MAXNM][MAXNM]){
    if(cnt == 3){
        memset(seen, false, sizeof(seen));
        ans = max(ans, bfs(a));
        return;
    }
    if(y ==m) x+=1, y = 0; // 실수(1m) : n으로 씀
    if(x ==n) return;
    rep(i, x, n){
        for(int j = y; j < m; j++, y=0){
            if(a[i][j] == 0){
                a[i][j] = 1;
                backtrack(i, j+1, cnt+1, a);
                a[i][j] = 0;
            }
        }
    }
}

void process(){
    input();
    all0 -= 3;
    int a[MAXNM][MAXNM];
    memcpy(a, oa, sizeof(a));
    backtrack(0, 0, 0, a);
    cout << ans << endl;
}
int main(){
    freopen("input.txt", "r", stdin);
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    process();
    return 0;
}
```

# 디버깅(10)
-  int a[8][8]; int oa[8][8]; 일때, 아래와 같은 memcpy는 작동하지 않습니다.
  - memcpy(a, oa, sizeof(oa));
- 문제를 찾느라 시간을 많이 소모하였고, 이럴 땐 아래와 같이 for문을 통해 초기화 시켜주어야 합니다.


```cpp
for(int i = 0; i < 8; i++)
    for(int j = 0; j < 8; j++)
        a[i][j] = oa[i][j];
```
