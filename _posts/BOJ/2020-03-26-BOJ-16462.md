## 실제 구현 

```cpp
#include<bits/stdc++.h>
#define endl '\n'
#define rep(i,a,b) for(int i=a;i<b;i++)
#define r_rep(i,a,b) for(int i=a;i>b;i--)
using namespace std;

int main(){
    freopen("input.txt", "r", stdin);
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    int n; cin >> n;
    int sum = 0;
    string x;
    rep(i, 0, n){
        cin >> x;
        rep(j, 0, x.size())
            if(x[j] == '0' || x[j] == '6')
                x[j] = '9';
        sum += min(100, stoi(x));
    }
    cout << round((double)sum/n) << endl;
    return 0;
}
```
