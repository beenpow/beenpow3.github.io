# 구현(10)
- queue를 이용한 위상 정렬 방법을 이용하였습니다.


```cpp
#include<bits/stdc++.h>
using namespace std;
const int MAX_N = 32000 + 1;

int n, m;
int indegree[MAX_N];
vector<int> adj[MAX_N];
queue<int> q; // idx, depth
void solve(){
    vector<int> ans;
    while(!q.empty()){
        int here = q.front(); q.pop();
        cout << here << " ";
        ans.push_back(here);
        for(int i = 0; i < adj[here].size(); i++){
            int there = adj[here][i];
            indegree[there]--;
            if(indegree[there] == 0){
                q.push(there);
            }
        }
    }
    cout << endl;
}
int main(){
    freopen("input.txt", "r", stdin);
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cin >> n >> m;
    while(m--){
        int x, y;
        cin >> x >> y;
        adj[x].push_back(y);
        indegree[y]++;
    }
    for(int i = 1; i <= n; i++)
        if(indegree[i] == 0)
            q.push(i);
    solve();
    return 0;
}
```
