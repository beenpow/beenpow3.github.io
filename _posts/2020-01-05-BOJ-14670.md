# BOJ::14670 병약한 영정
[BOJ]:<https://www.acmicpc.net/problem/14670>
- Level : Silver 3

# 시사점

- hash를 사용할 수 있는 문제입니다.
  - array사용법과 linkedList 사용법에 대해 각각 풀이하였습니다.

# 이해(3)

- 특정 약품 번호로 해결할 수 있는 병의 증세가 있습니다.
  - 이 값을 key와 value라는 이름으로 메모리에 할당하였습니다.

# 설계(5)

## 시간 복잡도

- Hash size 를 충분히 사용할 수 있으므로, 
  - table에서 값을 찾는 시간을 O(1)로 예상하여, 전체 시간 복잡도는 O(n)이라 할 수 있습니다.

# 구현(25)

## 배열 사용법

```cpp
#include <iostream>
using namespace std;

const int PN = 23;
const int HASH_SIZE = 10;
const int MAX_N = 100;

int name_size = 0;
int name[MAX_N][2];
int table[HASH_SIZE][MAX_N];

int n, r;

unsigned int getKey(unsigned int x){
    return (x * PN) % HASH_SIZE;
}

void insert(int key, int value){
    name[name_size][0] = key;
    name[name_size][1] = value;

    int hash_key = getKey(key);
    int& h_size = table[hash_key][0];
    table[hash_key][++h_size] = name_size;
    name_size++;
}
int search(int key){
    int ret = -1;
    int hash_key = getKey(key);
    int h_size = table[hash_key][0];
    for(int i = 1; i <= h_size; i++){
        int nameIdx = table[hash_key][i];
        if(name[nameIdx][0] == key){
            ret = name[nameIdx][1];
            break;
        }
    }
    return ret;
}
int main(){
    //freopen("input.txt", "r", stdin);
    ios_base::sync_with_stdio();
    cin.tie();
    cin >> n;
    for(int i = 0; i < n; i++){
        int a, b;
        cin >> a >> b;
        insert(a, b);
    }
    cin >> r;
    for(int i = 0; i < r; i++){
        int cnt, tmp;
        cin >> cnt;
        bool failed = false;
        int answer[100]={0};
        for(int j = 0; j < cnt; j++){
            cin >> tmp;
            answer[j] = search(tmp);
            if(answer[j] == -1) failed = true;
        }
        if(failed)cout << "YOU DIED" << endl;
        else{
            for(int j = 0; j < cnt; j++){
                if(j < cnt-1) cout << answer[j] << " ";
                else cout << answer[j];
            }
            cout << endl;
        }
    }
    return 0;
}
```

## 링크드 리스트 사용법

```cpp
#include <iostream>
using namespace std;

const int HASH_SIZE = 1000;
const int MAX_N = 100;

int n, r;

unsigned int getKey(unsigned int x){
    return (x) % HASH_SIZE;
}

struct node{
    int key;
    int value;
    struct node* next;
};
struct bucket{
    int count;
    struct node* head;
    struct node* tail;
};

struct bucket* bucket[HASH_SIZE];

void insert(int key, int value){
    int bucket_key = getKey(key);
    struct node* newNode = new struct node();
    newNode->key = key;
    newNode->value = value;
    newNode->next = NULL;

    if(bucket[bucket_key]->tail == NULL){
        bucket[bucket_key]->head = newNode;
        bucket[bucket_key]->tail = newNode;
    }else{
        bucket[bucket_key]->tail->next = newNode;
        bucket[bucket_key]->tail = newNode;
    }
    bucket[bucket_key]->count++;
}

int search(int key){
    int ret = -1;
    int bucket_key = getKey(key);
    struct node* horse = bucket[bucket_key]->head;
    for(int i = 0; i < bucket[bucket_key]->count; i++){
        if(horse->key == key){
            ret = horse->value;
            break;
        }
        horse = horse->next;
    }
    return ret;
}

int main(){
    freopen("input.txt", "r", stdin);
    ios_base::sync_with_stdio();
    cin.tie();
    cin >> n;
    for(int i = 0; i < HASH_SIZE; i++){
        bucket[i] = new struct bucket();
        bucket[i]->count = 0;
        bucket[i]->head = NULL;
        bucket[i]->tail = NULL;
    }
    for(int i = 0; i < n; i++){
        int a, b;
        cin >> a >> b;
        insert(a, b);
    }
    cin >> r;
    for(int i = 0; i < r; i++){
        int cnt, tmp;
        cin >> cnt;
        bool failed = false;
        int answer[100]={0};
        for(int j = 0; j < cnt; j++){
            cin >> tmp;
            answer[j] = search(tmp);
            if(answer[j] == -1) failed = true;
        }
        if(failed)cout << "YOU DIED" << endl;
        else{
            for(int j = 0; j < cnt; j++){
                if(j < cnt-1) cout << answer[j] << " ";
                else cout << answer[j];
            }
            cout << endl;
        }
    }
    return 0;
}

```

# 디버깅(5)

- 링크드 리스트의 사용이 아직 미숙합니다.
  - 특히, tail과 head를 사용하는 것이 미숙하여 시간이 소모되었습니다.
