```cpp
#include<bits/stdc++.h>
#define endl '\n'
#define rep(i,a,b) for(int i=a;i<b;i++)
using namespace std;
const int seq[]={3, 6, 6, 2, 4, 0, 2, 5, 1, 3, 6, 1};
const string ans[]={"Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"};
int main(){
    freopen("input.txt", "r", stdin);
    cin.tie(0);
    int a, b;
    cin >> b >> a ;
    b = (b-1)%7;
    int stDay = seq[a-1];
    int st = 0;
    while(true){
        if(st == b){
            cout << ans[stDay] << endl;
            break;
        }
        st++;
        stDay = (stDay+1)%7;
    }
    return 0;
}
```
