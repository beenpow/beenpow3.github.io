## sort
- 박트리님의 블로그를 참고하였습니다.


```cpp
#include <stdio.h>
#include <iostream>
#include <vector>
using namespace std;


template <typename It>
void _swap(It &a, It &b){
	It c(a);a=b;b=c;
}
template <typename It, typename Comp>
void _sort(It begin, It end, Comp comp){
	if(begin==end)
		return;
	_swap(*begin, *((end-begin)/2+begin));
	It pi = begin;
	It le = begin + 1;
	It ri = end - 1;
	while( le <= ri ) {
		while(le <= ri && !comp(*pi, *le))
			le++;
		while(le <= ri && !comp(*ri, *pi))
			ri--;
		if(le<=ri)
			_swap(*le, *ri);
	}
	_swap(*pi, *ri);
	_sort(begin, ri, comp);
	_sort(ri + 1,  end, comp);
}

template <typename A, typename B>
struct _pair{
	A first;
	B second;
	_pair(A a, B b):first(a), second(b){}
	_pair(){}
};

bool cmp(const _pair<int, int>& a, const _pair<int, int>& b){
	if(a.first==b.first)
		return a.second<b.second;
	return a.first<b.first;
}
// __cmp
bool __cmp(const int& a, const int& b){
    return a < b;
}

int main(){
	vector<_pair<int, int> > arr;
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++){
		int a, b;
		scanf("%d%d", &a, &b);
		arr.push_back({a, b});
	}
	_sort(arr.begin(), arr.end(), cmp);
	for(int i=0;i<n;i++)
		printf("%d %d\n", arr[i].first, arr[i].second);
	return 0;
}
```

```cpp
#include<bits/stdc++.h>
using namespace std;

void _swap(int& A, int& B){int c =A; A = B; B = c;}
void _qsort(int A[], int begin, int end){
    if(begin >= end) return;
    int i = begin-1, j = begin;
    int& pivot = A[end];
    for(;j < end; j++)
        if(A[j] < pivot)
            _swap(A[++i], A[j]);
    _swap(A[++i], pivot);
    _qsort(A, begin, i-1);
    _qsort(A, i+1, end);
}

struct cells{
    int first;
    int second;
}cell;
void __swap(struct cells& A, struct cells& B){struct cells C = A; A = B; B = C;}
void __qsort(struct cells B[10], int begin, int end){
    if(begin >= end) return;
    int i = begin-1, j = begin;
    struct cells& pivot = B[end];
    for(; j < end; j++)
        if((B[j].first < pivot.first) || ((B[j].first == pivot.first) && (B[j].second < pivot.second)))
            __swap(B[++i], B[j]);
    __swap(B[++i], pivot);
    __qsort(B, begin, i-1);
    __qsort(B, i+1, end);
}
```

# merge sort

## merge sort ( no cmp )

```cpp
int a[MAXN], b[MAXN];
void input() {
	ans = 0;
	cin >> n;
	rep(i, 1, n + 1) {
		cin >> a[i];
		b[i] = 0;
	}
}
// [le ... mid] [mid+1 ...  ri]
// id3 will be the index of b
void merge(int le, int mid, int ri) {
	int id1 = le, id2 = mid + 1, id3 = le;
	while (id1 <= mid && id2 <= ri) {
		if (a[id1] <= a[id2])
			b[id3++] = a[id1++];
		else {
			ans += (mid - id1 + 1);
			b[id3++] = a[id2++];
		}
	}

	if (id1 <= mid) {
		rep(i, id1, mid + 1) b[id3++] = a[i];
	}
	else {
		rep(i, id2, ri + 1) b[id3++] = a[i];
	}
	rep(i, le, ri + 1) a[i] = b[i];
}
void mergesort(int le, int ri) {
	if (le < ri) {
		int mid = (le + ri) / 2;
		mergesort(le, mid);
		mergesort(mid + 1, ri);
		merge(le, mid, ri);
	}
}
void process() {
	input();
	mergesort(1, n);
	cout << ans << endl;
}
```

## merge sort( with cmp )

```cpp
#define rep(i, a, b) for(int i=(a);i<(b);i++)
#include<iostream>
using namespace std;
typedef struct point{
    int x, y;
}point;

int n;
point list[100000], sorted[100000];

int cmp(const point a, const point b){
    if(a.x == b.x) return a.y < b.y;
    else return a.x < b.x;
}
void merge(point list[], int le, int mid, int ri, int (*cmp)(point, point)){
    int i = le, j = mid+1, k = le;
    while(i <= mid && j <= ri){
        if(cmp(list[i], list[j]))
            sorted[k++] = list[i++];
        else
            sorted[k++] = list[j++];
    }
    
    if(i <= mid){
        rep(l, i, mid+1) sorted[k++] = list[l];
    }else{
        rep(l, j, ri+1) sorted[k++] = list[l];
    }
    rep(l, le, ri+1) list[l] = sorted[l];
}
void msort(point list[], int le, int ri, int (*cmp)(point, point)){
    if(le < ri){
        int mid = (le + ri) / 2;
        msort(list, le, mid, cmp);
        msort(list, mid+1, ri, cmp);
        merge(list, le, mid, ri, cmp);
    }
}
int main() {
    freopen("input.txt", "r", stdin);
    rep(i, 0, 10){
        int x, y;
        cin >> x >> y;
        list[i] = {x, y};
    }
    // 1. get input to list
    // 2. merge sort
    msort(list, 0, 10 - 1, cmp);
    
    rep(i, 0, 10){
        cout << list[i].x <<" " << list[i].y << endl;
    }
    return 0;
}
```
