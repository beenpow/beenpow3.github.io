# Queue 정의
- 박트리님의 블로그를 참고하였습니다.


```cpp
#include <bits/stdc++.h>
using namespace std;

template <class T>
class _queue {
public:
	struct Node {
		T val;
		Node *next;
		Node() {}
		Node(T val) :val(val), next(0){}
	};
	Node *head;
	Node *tail;
	int _size;
	_queue() {
		head = tail = 0;
		_size = 0;
	}
	void push(T val) {
		Node *temp = new Node(val);
		if (head == 0)
			head = tail = temp;
		else {
			tail->next = temp;
			tail = temp;
		}
		_size++;
	}
	void pop() {
		Node *temp = head;
		head = head->next;
		delete temp;
		_size--;
	}
	bool empty() const{
		return _size == 0;
	}
	T front() const{
		return head->val;
	}
	T back() const{
		return tail->val;
	}
	int size() const{
		return _size;
	}
};
```

# priority_queue 정의
- heap 을 이용한 pq의 구현입니다.
  - heap은 vector를 사용하여 구현하였습니다.
  - 정적 메모리 할당시 memory limit을 초과할 수 있습니다.

```cpp
typedef struct node{
    int time;
}node;

template <typename T>
class _priority_queue {
    int _size;
    _vector <node> heap;
    //toolNode heap[MAX_LOT];
public:
    _priority_queue() {}
    ~_priority_queue() { _size = 0, heap.clear(); }
    void clear() {
        _size = 0;
        heap.clear();
    }
    void push_heap(int TIME) {
        heap.push_back({ TIME });
        _size++;
        int idx = _size - 1;
        while (idx > 0 && heap[(idx - 1) / 2].T < heap[idx].time) {
            _swap(heap[idx], heap[(idx - 1) / 2]);
            idx = (idx - 1) / 2;
        }
    }
    bool isempty() { return _size == 0; }
    node top() {
        node ret = heap[0];
        return ret;
    }
    int size(){return _size;}
    node pop_heap() {
        node ret = heap[0];
        heap[0] = heap[_size - 1];
        heap.pop_back();
        _size -= 1;
        int here = 0;
        while (true) {
            int left = here * 2 + 1, right = here * 2 + 2;
            // when it gets to leaf
            if (left >= _size) break;
            int next = here;
            if (heap[next].time < heap[left].time)
                next = left;
            if (right < _size && heap[next].time < heap[right].time)
                next = right;
            if (next == here) break;
            _swap(heap[here], heap[next]);
            here = next;
        }

        return ret;
    }
};
```
